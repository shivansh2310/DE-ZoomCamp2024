> For pg-network

winpty docker run -it \
    -e POSTGRES_USER="root" \
    -e POSTGRES_PASSWORD="root" \
    -e POSTGRES_DB="ny_taxi" \
    -v D://Data_Engg//DE-ZoomCamp2024//week_1//ny_taxi_postgres_data:/var/lib/postgresql/data \
